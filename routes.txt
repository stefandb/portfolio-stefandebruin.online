admin
Route::prefix('admin')
    ->middleware(['auth'])
    ->name('admin.')
    ->group(function () {

        Route::get('/', DashboardController::class)->name('dashboard');

        Route::resource('projects', AdminProjectController::class);
        Route::resource('posts', AdminPostController::class);

        Route::post('/admin/uploads', UploadController::class)
            ->middleware('auth');
    });
